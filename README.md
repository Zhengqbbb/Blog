```bash
# 安装hexo
npm install -g hexo-cli
npm install -g hexo-cli --registry=http://registry.npm.taobao.org
# 运行本地环境
hexo server
# ... 写文章 source/_posts
# 清楚上一次打包资源
hexo clean
# 生成静态文件
hexo generate
# 推送到仓库
hexo deploy
```

```bash
# 自定义脚本
# 运行本地环境
npm run dev
# 一条命令推送到仓库
npm run push
```
